    def __init__(self):
        self.waypoints_printed = False

            self.waypoints_printed = True
            print(f"<<<waypoints_start>>>{waypoints}<<<waypoints_end>>>")
        print(f"<<<params_start>>>{params}<<<params_end>>>")
